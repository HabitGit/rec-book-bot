Users module:

registration:
    /registration
        req:
        POST {
                userId: 1,
                chatId: 1,
                name: name,
                inviteLink: link
             }
        res:
        STATUS 201 - created
        bad res:
        400 - Невалидные данные
        422 - Такой юзер уже зарегистрирован
        500 - Ошибка сервера

Add friend:
    /profile/:id/friends
    req:
    POST: {
             friendId: 1
          }
    res:
    STATUS: 201 - create
    bad res:
    400 - Bad request,
    422 - Такой юзер не зарегистрирован
    401 - unauthorized,
    500 - server error

My books/friends:
